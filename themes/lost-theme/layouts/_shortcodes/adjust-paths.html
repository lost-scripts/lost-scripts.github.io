{{- /* Smart HTML path fixer for multi-language bundles (Handles 'assets/' and './assets/') */ -}} {{- $content := .Inner -}} {{- $isTranslation := false -}} {{- /* Check if the current page URL belongs to a translation */ -}} {{- if or (strings.Contains .Page.RelPermalink "/es/") (strings.Contains .Page.RelPermalink "/ca/") -}} {{- $isTranslation = true -}} {{- end -}} {{- if not $isTranslation -}} {{- /* Default language: leave assets as-is for GitHub and native rendering */ -}} {{- $content | safeHTML -}} {{- else -}} {{- /* Translation: inject the 3-level-back path */ -}} {{- $scriptDir := urlize (path.Base .Page.File.Dir) -}} {{- $content = $content | replaceRE `(src|href|srcset)=['"](?:\./)?assets/([^'"]+)['"]` (printf "$1=\"../../../scripts/%s/assets/$2\"" $scriptDir) -}} {{- $content | safeHTML -}} {{- end -}}
